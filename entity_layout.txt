int asset_group_count
{
    byte[16] module_id
    byte[16] asset_type_id
    int asset_count
    {
        byte[16] asset_id
    }[asset_count] assets
}[asset_group_count] asset_groups

magic_word 0xCCBBFFF2

int entity_count
{
    // no longer includes position/transform in entity
    int entity_id
    int parent_entity
}[entity_count] entities

magic_word 0xCCBBFFF3

int component_group_count
{
    byte[16] module_id 
    byte[16] component_type_id
    int component_count
    {
        int component_id
        int entity_id
        int field_count
        {
            byte[16] field_name
            Variant field_value
        }[field_count] fields
    }[component_count] components
}[component_group_count] component_groups
